string对象

charAt() 返回指定位置的字符

								
				var str="wefrwef"
				document.write(str.charAt(1))
              输出为e

concat() 链接字符串

						var str1="hch"
						var str2="yyy"
					document.write(str1.concat(str2))
					
					
					输出为hchyyy

indexOf() 检索字符串

			       var str="fwfwfw"
			       document.write(str.indexOf("w"))
			       
			       
			       输出为1，从0开始检索，返回第一个出现的位置

match()在字符串内检索指定的值和indexOf()差不多但是返回的是值不是位置

			var str="hihihch"
			document.write(str.match("hi"))
			输出为hi
			document.write(str.match("gg"))
			输出为null

slice提取字符串的某个部分

		 	       var str="huang"
		 	       document.write(str.slice(2))
		 	       输出为ang,不改变原有字符串

split()把字符串分割成字符串数组

		 	 var str="huang:chen:hui"
		 	 document.write(str.split(":"))
		 	 输出为huang,chen,hui

array对象

pop()删除数组的最后一个元素，并返回最后一个元素

		 	 var arr=["huang","chen","hui"]
		 	 document.write(arr.pop())
		 	 输出为hui
		 	 document.write(arr)
		 	 输出为huang,chen

push()给数组尾部添加一个元素,并返回新的长度

		 	    var arr=["huang","chen","hui",];
		 	    document.write(arr.push("cc"));
		 	    输出为4
		 	   document.write(arr)
		 	   输出为huang,chen,hui,cc

shift()删除数组的第一个元素，并返还第一个元素

		 	   var arr=["huang","chen","hui"]
		 	   document.write(arr.shift())
		 	   输出为hui
		 	   document.write(arr)
		 	   输出为huang,chen

unshift()给数组开头添加一个元素，并返回新的长度

		 	  var arr=["huang","chen","hui"]
		 	  document.write(arr.unshift("cc"))
		 	  输出为4
		 	  document.write(arr)
		 	  输出为cc,huang.chen,hui
